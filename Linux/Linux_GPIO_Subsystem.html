1. 架构总览 — GPIO 子系统的三层模型

GPIOLIB 核心 · gpio_chip 驱动接口 · gpiod_* 消费者 API — 从硬件到应用的完整数据路径

层级角色核心数据结构关键文件

gpio_chip — 控制器抽象

include/linux/gpio/driver.h

GPIO 控制器驱动必须填充的回调集。get_direction / direction_input / direction_output / get / set 为 6 个核心 ops。ngpio 声明管理的引脚数，can_sleep 标记是否可睡眠 (I2C/SPI 扩展器 = true)。通过 devm_gpiochip_add_data() 注册。

gpio_desc — 单线描述符

gpiolib.h (私有)

每个物理 GPIO 线对应一个 gpio_desc。包含指向 gpio_chip 的指针、FLAG_REQUESTED / FLAG_IS_OUT / FLAG_OPEN_DRAIN / FLAG_USED_AS_IRQ 等状态位、label 名称字符串。2024 年将 debounce_period_us 移出以缩小结构体。

gpio_device — 设备抽象 (NEW)

2023-2024 重构的封装层

2024 年重大封装改进: gpio_device 替代直接访问 gpio_chip。包含全局 ID、嵌入的 device (sysfs)、cdev (字符设备)、SRCU 保护的 chip 指针。新 API: gpio_device_find_by_label/fwnode() 替代已删除的 gpiochip_find()。

/* ═══ GPIO 注册路径 ═══ */ gpio-mychip.c (驱动) ├── 填充 gpio_chip │ ├── .label = "mygpio" │ ├── .ngpio = 32 │ ├── .get_direction / .direction_input / .direction_output │ ├── .get / .set │ └── 可选: .to_irq / .set_config / .set_debounce └── devm_gpiochip_add_data(dev, chip, priv) └── gpiochip_add_data_with_key() ├── 分配 gpio_device + 内嵌 device + cdev ├── 分配 gpio_desc[ngpio] 数组 ├── 每个 desc 反向指向 chip ├── 注册字符设备 /dev/gpiochipX ├── 注册 sysfs 属性 └── 加入全局 SRCU 保护的设备链表

2. GPIOLIB 2024 年度重构 — SRCU 并发 · 封装 · 废弃清理

Bartosz Golaszewski 主导的最大重构 — 全局自旋锁退役 · gpio_chip 封装 · API 现代化

并发模型升级: spinlock → SRCU

gpio_lock 自旋锁被彻底移除

旧设计中全局 gpio_lock spinlock 保护设备链表和描述符字段 → 所有 GPIO 操作竞争同一把锁。新设计使用 SRCU (Sleepable RCU): 读取方无锁并发，写入方 (注册/注销) 使用 synchronize_srcu() 等所有读取方退出。描述符 label、chip 指针等字段均受 SRCU 保护。

删除/废弃的旧 API

v6.8 – v6.11 清理清单

gpiochip_add() — 已删除，统一用 gpiochip_add_data()
gpiochip_find() — 已删除，改用 gpio_device_find()
gpiochip_is_requested() — 已删除，改用 gpiochip_dup_line_label()
gpio_cansleep() — 已删除
gpiochip_get_desc() — 已 unexport (内部函数)
IRQ domain 函数 — 全部 unexport 为内部专用

3. Consumer API — 驱动开发者如何使用 GPIO

描述符获取 → 方向设置 → 读写值 → 中断转换 — 完整的驱动侧 GPIO 操作手册

/* ===== Consumer API 完整示例 (include/linux/gpio/consumer.h) ===== */ /* 1. 获取 GPIO 描述符 */ struct gpio_desc *reset_gpio; // 不透明句柄 — 不再使用 int 编号 struct gpio_desc *led_gpio; struct gpio_descs *data_bus; // 批量 GPIO // 从设备树 "reset-gpios" 属性获取 (推荐 devm 托管版本) reset_gpio = devm_gpiod_get(dev, "reset", GPIOD_OUT_HIGH); led_gpio = devm_gpiod_get_optional(dev, "led", GPIOD_OUT_LOW); // 按索引获取 (如 "gpios" = <&gpio 5 0, &gpio 6 0>) data_0 = devm_gpiod_get_index(dev, "data", 0, GPIOD_IN); data_1 = devm_gpiod_get_index(dev, "data", 1, GPIOD_IN); // 批量获取 (一把拿所有) data_bus = devm_gpiod_get_array(dev, "data", GPIOD_IN); /* 2. 方向与值操作 */ gpiod_direction_input(desc); // 设为输入 gpiod_direction_output(desc, 1); // 设为输出, 初始值=1 int val = gpiod_get_value(desc); // 读电平 (0/1) gpiod_set_value(desc, 0); // 写电平 /* 3. 高级配置 */ gpiod_set_debounce(desc, 5000); // 去抖 5ms gpiod_set_open_drain(desc); // 开漏模式 gpiod_set_open_source(desc); // 开源模式 /* 4. 中断 — GPIO 线作为 IRQ 源 */ int irq = gpiod_to_irq(desc); // GPIO → IRQ 编号 request_irq(irq, my_handler, IRQF_TRIGGER_RISING, "myirq", dev); /* 5. 释放 — devm 版本自动释放, 手动版本: */ gpiod_put(desc);

4. pinctrl — 引脚复用与电气配置

GPIO 只是众多引脚功能之一 · pinctrl 负责决定"这个引脚是什么"· GPIO 操作其电平

pinctrl 管理的三个维度

Pin · Pin Group · Pin Function

Pin (引脚): 单个物理 PAD — 通过 pinctrl_pin_desc 描述
Pin Group (引脚组): 一组功能相关的引脚。如 SPI0 的 MOSI/MISO/CLK/CS 四条线
Pin Function (功能): 引脚的 ALT 模式。如 "gpio", "spi0", "uart2_tx", "i2c0_scl"
Pin Configuration (电气): bias-pull-up/down、drive-strength、slew-rate、schmitt-trigger

GPIO 与 pinctrl 的桥接

gpio_request 隐式调用 pinctrl

gpiod_get() → gpiod_request() → pinctrl_gpio_request() → pinmux_ops.gpio_request_enable() — 将引脚复用切换为 GPIO 模式。
反过来，consumer 设备的 pinctrl-0 属性在驱动 probe 时自动应用，将引脚切换到对应外设功能。
两个子系统自动协同 — 驱动开发者通常无需手动管理。

pinctrl 状态何时应用典型用途
"default"驱动 probe 时自动正常工作模式
"sleep"系统 suspend低功耗引脚配置
"idle"runtime PM idle空闲省电但不全休眠
"init"probe 前 (先于 default)硬件初始化状态

6. GPIO 驱动开发 — Provider 侧完整模板

从零编写 gpio_chip 驱动 · devm 托管 · IRQ chip 集成 · 2024 推荐实践

/* ===== GPIO 控制器驱动 — 最小实现 (platform_driver) ===== */ #include <linux/gpio/driver.h> #include <linux/platform_device.h> struct my_gpio_priv { struct gpio_chip chip; // 嵌入 gpio_chip void __iomem *base; // 寄存器基址 spinlock_t lock; // 寄存器访问锁 }; /* ===== gpio_chip 核心回调实现 ===== */ static int my_gpio_get(struct gpio_chip *gc, unsigned int offset) { struct my_gpio_priv *priv = gpiochip_get_data(gc); u32 val = readl(priv->base + GPIO_DATA_IN); return !!(val & BIT(offset)); } static void my_gpio_set(struct gpio_chip *gc, unsigned int offset, int value) { struct my_gpio_priv *priv = gpiochip_get_data(gc); unsigned long flags; spin_lock_irqsave(&priv->lock, flags); if (value) writel(BIT(offset), priv->base + GPIO_DATA_SET); else writel(BIT(offset), priv->base + GPIO_DATA_CLR); spin_unlock_irqrestore(&priv->lock, flags); } static int my_gpio_direction_input(struct gpio_chip *gc, unsigned int offset) { /* 写方向寄存器: 0 = 输入 */ return 0; } static int my_gpio_direction_output(struct gpio_chip *gc, unsigned int offset, int v) { /* 写方向寄存器: 1 = 输出, 再写值 */ return 0; } static int my_gpio_get_direction(struct gpio_chip *gc, unsigned int offset) { /* 读方向寄存器 */ return GPIO_LINE_DIRECTION_IN; } /* ===== probe — 注册 gpio_chip ===== */ static int my_gpio_probe(struct platform_device *pdev) { struct my_gpio_priv *priv; struct gpio_chip *gc; priv = devm_kzalloc(&pdev->dev, sizeof(*priv), GFP_KERNEL); priv->base = devm_platform_ioremap_resource(pdev, 0); spin_lock_init(&priv->lock); gc = &priv->chip; gc->label = "mygpio"; // 显示名 gc->ngpio = 32; // 管理的线数 gc->parent = &pdev->dev; // 父设备 (sysfs 关联) gc->base = -1; // 动态分配基址 (推荐) gc->can_sleep = false; // MMIO 寄存器访问不同睡眠 gc->get_direction = my_gpio_get_direction; gc->direction_input = my_gpio_direction_input; gc->direction_output = my_gpio_direction_output; gc->get = my_gpio_get; gc->set = my_gpio_set; return devm_gpiochip_add_data(&pdev->dev, gc, priv); } static struct platform_driver my_gpio_drv = { .driver = { .name = "my-gpio", .of_match_table = my_match_table }, .probe = my_gpio_probe, }; module_platform_driver(my_gpio_drv);

7. 用户空间交互 — gpio-tools 与字符设备

CDev /dev/gpiochipX · gpiodetect/gpioinfo/gpioget/gpioset/gpiomon · 废弃的 sysfs /sys/class/gpio

字符设备接口 (现代推荐)

/dev/gpiochipX — ioctl + poll

gpiodetect — 列出所有 GPIO 控制器
gpioinfo — 查看控制器所有线的状态 (consumer/方向/电平)
gpioget — 读一条线 (自动 request→读→release)
gpioset — 写一条线 (支持保持模式: 退出后不释放)
gpiomon — 监控边沿事件 (上升/下降/双沿 + 时间戳)

旧 sysfs 接口 (已废弃)

/sys/class/gpio/ — 勿用于新设计

echo N > /sys/class/gpio/export — 导出 GPIO N
echo out > /sys/class/gpio/gpioN/direction
echo 1 > /sys/class/gpio/gpioN/value
问题: 全局编号不稳定、无所有权追踪、已导出 GPIO 拔模块时内核 oops。字符设备已完全覆盖所有功能且更安全。

/* ===== gpio-tools 实战 ===== */ # 列出所有 GPIO 控制器 gpiodetect # gpiochip0 [gpio0] (32 lines) # gpiochip1 [gpio1] (32 lines) # gpiochip2 [pca953x] (16 lines) # 查看 gpiochip0 所有线的详细信息 gpioinfo 0 # line 0: "LED0" output active-high [used] # line 1: unnamed input active-high [unused] # 读一条 GPIO gpioget 0 1 // gpiochip0 line 1 # 写一条 GPIO (保持模式: 退出后不释放) gpioset --hold 0 0=1 // gpiochip0 line 0 = high, 不释放 # 监控边沿事件 gpiomon 0 15 // 监控 gpiochip0 line 15 的双沿事件 # event: RISING EDGE offset: 15 timestamp: [1730123456.789012]
字符设备 IOCTL 支持的高级操作
Line Handle: 一次取多条线 → 原子性读写 → 适合并行总线控制
Line Event: 边沿检测 + 实时时间戳 (CLOCK_MONOTONIC / CLOCK_REALTIME) → 无需轮询
Watch 模式: 内核主动通知线信息变更 (consumer 请求/释放) → gpio-watch 工具
